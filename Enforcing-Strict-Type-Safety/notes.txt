paths:{"@/*":["./src/*"]}
    giving an alias to the path in the array

noUnusedLocals: true
    unused local variables will be marked and show warning

noUnusedParamters" true
    unused function parameters will be marked and show warning

allowUnusedLabels: false
    c/c++ like go-to statements not allowed

noUncheckedSideEffectImports: true
    importing a file which doesn't exist will show error

noFallThroughCasesInSwitch: true
    avoiding executing next cases in switch after executing the required one

allowUnreachableCode: false
    show warning when unreachable code is there

noUncheckedIndex: true
    - shows warning when direct access to array elements with index(even if index in range). increases code quality!!
    - number[index].toString() -> number[index]?.toString()

noPropertyAccessFromIndexSignature: true
    - doesn't allows to use a not-defined property when using '.' notation, - useful in case of generic defined properties

erasableSyntaxOnly: true
    - dont allow giving access modifier in function args
    - function(private value:number){} //will give error

noImplicitOverride: true
    dont allow overriding a method silently, it must be explicitly defined

noErrorTruncation: true
    show more available info about the error

exactOptionalPropertyTypes: true
    - for optional property, dont allow assigning undefined
    - {prop?:string}, obj.prop = undefined //valid by default

allowJs: true
    allow importing js file in ts file

checkJs: true
    check types in js file

@ts-check(put it at top and only for js file)
    - check types in js file
    - useful in case of lazy migration to ts
